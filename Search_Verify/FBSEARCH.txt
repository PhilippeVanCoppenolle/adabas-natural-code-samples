/* OPTIONS MCG=(ON) /* use NOC if possible
DEFINE FUNCTION FBSEARCH
  RETURNS RESULT (I4)
  DEFINE DATA
  PARAMETER
    1 #P_STRING   (B) DYNAMIC BY VALUE
    1 #P_SEARCH   (B) DYNAMIC BY VALUE
  LOCAL
    1 #I  (I4)
    1 #J  (I4)
    1 #B1 (B1)
  END-DEFINE
*
RESULT := 0
F1. FOR #I := 1 TO *LENGTH(#P_STRING)
    #B1 := SUBSTR(#P_STRING, #I, 1)
    EXAMINE FULL #P_SEARCH FOR FULL #B1 GIVING POSITION IN #J
    IF #J <> 0 THEN
      RESULT := #I
      ESCAPE BOTTOM (F1.)
    END-IF
  END-FOR /* (F1.)
END-FUNCTION
END
